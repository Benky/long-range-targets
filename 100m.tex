% 1MoA = ((2*Pi*100)/(360*60))*100
\newcommand{\OneMOA}{29.08882086657216mm}

\newcommand{\horizontalStepSize}{\OneMOA}
\newcommand{\verticalStepSize}{\OneMOA}

\newcommand{\horizontalCellsNum}{9}
\newcommand{\verticalCellsNum}{6}

% Red diamond with cross
\newcommand\benkysCross{
  \draw[line width=3.5mm] (-1 * \OneMOA, 0)--(1 * \OneMOA ,0);
  \draw[line width=3.5mm] (0, -1 * \OneMOA)--(0, 1 * \OneMOA);
  \draw[color=black] (0.5in ,0.5in) rectangle (-0.5in , -0.5in);
  \begin{scope}
    \draw[color=black, fill=red, rotate around={45:(0, 0)}] (\OneMOA /4 ,\OneMOA /4) rectangle (-\OneMOA /4 , -\OneMOA /4);
  \end{scope}
}
